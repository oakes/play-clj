(defn tree*
  [children arg]
  (create-group (Tree. arg) children))
(defmacro tree
  [children arg & options]
  `(let [entity# (tree* ~children ~arg)]
     (u/calls! ^Tree (u/get-obj entity# :object) ~@options)
     entity#))
