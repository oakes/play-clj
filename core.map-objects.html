(defn map-objects*
  ([]
    (MapObjects.))
  ([^MapLayer layer]
    (.getObjects layer)))
(defmacro map-objects
  [^MapLayer layer & options]
  `(let [^MapObjects object# (map-objects* ~layer)]
     (u/calls! object# ~@options)))
