(defn animation-controller*
  [entity]
  (AnimationController. (u/get-obj entity :object)))
(defmacro animation-controller
  [entity & options]
  `(let [^AnimationController object# (animation-controller* ~entity)]
     (u/calls! object# ~@options)))
