(defn container*
  [child]
  (ActorEntity. (Container. (u/get-obj child :object))))
(defmacro container
  [child & options]
  `(let [entity# (container* ~child)]
     (u/calls! ^Container (u/get-obj entity# :object) ~@options)
     entity#))
